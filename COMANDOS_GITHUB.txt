# COMANDOS PARA SUBIR A GITHUB - COPIA Y PEGA UNO POR UNO

# 1. Configura tu identidad (CAMBIA los datos por los tuyos)
git config --global user.name "Tu Nombre Completo"
git config --global user.email "tu-email@gmail.com"

# 2. Haz el commit
git commit -m "Initial commit: RunTogether TFG - Frontend + Backend completo"

# 3. Crea el repositorio en GitHub
# Ve a: https://github.com/new
# Nombre: RunTogether-TFG
# NO marques "Add README" ni ".gitignore"
# Click "Create repository"

# 4. Conecta con GitHub (CAMBIA "TU-USUARIO" por tu usuario de GitHub)
git remote add origin https://github.com/TU-USUARIO/RunTogether-TFG.git

# 5. Sube el código
git push -u origin main

# Si te pide contraseña, necesitas un Personal Access Token:
# https://github.com/settings/tokens
# Genera uno nuevo y úsalo como contraseña
